\section{Délai de rendez-vous médecins}

\begin{enumerate}
  
  \item Faire un dataset avec toutes les données avec un champ pour le type de 
  médecins 
  \item rationnaliser la variable : sans supplémnent -> 0 euros
  \item Calculer le temps moyens d'attente pour chaque spécialité
  \item Tracer le temps d'attente vs tarifs pour les ophtalmo
  \item Calculer le prix moyen, le délai moyen par ville et spécialité
  \item Représentation graphiques des villes selon le délai et le tarif pour les 
  ophtalmologue (en abscisse le délai moyen, en ordonnée le tarif moyen et le nom 
  de la ville à place d'un point)
  \item Calculer la corrélation entre le temps d'attente moyen et le tarif moyen pour les ophtalmo
  \item Représentation graphiques des villes selon le délai et le tarif pour les 
  ophtalmologue pour les deux spécialités (faire le même graphique avec les deux 
  spécialités et des couleurs différentes)
  
\end{enumerate}

\clearpage
\section{Faire un dataset avec toutes les données avec un champ pour le type de 
  médecins}

\begin{knitrout}\footnotesize
\definecolor{shadecolor}{rgb}{0.969, 0.969, 0.969}\color{fgcolor}\begin{kframe}
\begin{alltt}
\hlstd{> }\hlkwd{require}\hlstd{(readxl)}
\hlstd{> }\hlstd{ophtalmo} \hlkwb{<-} \hlkwd{read_excel}\hlstd{(}\hlstr{"data/delai_rdv/Delai_rdv_ophtalmos_2012.xlsx"}\hlstd{)}
\hlstd{> }\hlstd{ophtalmo}\hlopt{$}\hlstd{spécialité} \hlkwb{<-} \hlstr{"ophtalmo"}
\hlstd{> }
\hlstd{> }\hlstd{gynecos} \hlkwb{<-} \hlkwd{read_excel}\hlstd{(}\hlstr{"data/delai_rdv/Delai_rdv_gynecos_2012.xlsx"}\hlstd{)}
\hlstd{> }\hlstd{gynecos}\hlopt{$}\hlstd{spécialité} \hlkwb{<-} \hlstr{"gynecos"}
\hlstd{> }
\hlstd{> }\hlstd{medecins} \hlkwb{<-} \hlkwd{rbind}\hlstd{(ophtalmo,gynecos)}
\end{alltt}
\end{kframe}
\end{knitrout}


\section{Calculer le temps moyens d'attente pour chaque spécialité}

\end{kframe}
\end{knitrout}


\section{Tracer le temps d'attente temps d'attente vs tarifs}

\begin{knitrout}\footnotesize

\begin{alltt}

\hlstd{+ }     \hlkwc{xlab}\hlstd{=}\hlstr{"Tarifs (euros)"}\hlstd{,}\hlkwc{ylab}\hlstd{=}\hlstr{"Délai (jours)"}\hlstd{,} \hlkwc{main}\hlstd{=}\hlstr{"Délai d'attente et tarifs"}\hlstd{)}

\end{knitrout}

\section{Calculer le prix moyen, le délai moyen par ville et spécialité}

\begin{knitrout}\footnotesize
\definecolor{shadecolor}{rgb}{0.969, 0.969, 0.969}\color{fgcolor}\begin{kframe}
\begin{alltt}
\hlstd{> }\hlkwd{colnames}\hlstd{(medecins)}
\hlstd{> }\hlstd{medecins}\hlopt{$}\hlstd{Délai} \hlkwb{<-} \hlstd{medecins}\hlopt{$}\hlstd{`Délai rendez-vous`}
\hlstd{> }\hlstd{med} \hlkwb{<-} \hlkwd{ddply}\hlstd{(medecins,} \hlkwd{.}\hlstd{(Ville,spécialité), summarise,}
\hlstd{+ }             \hlkwc{Délai} \hlstd{=} \hlkwd{mean}\hlstd{(Délai),} \hlkwc{Tarif} \hlstd{=} \hlkwd{mean}\hlstd{(Tarif))}
\hlstd{> }\hlstd{med.old} \hlkwb{<-} \hlstd{med}
\end{alltt}
\end{kframe}
\end{knitrout}

\section{Représentation graphiques des villes selon le délai et le tarif pour les 
  ophtalmologue}

Le but est de faire deux axes, un avec le tarif l'autre avec le délai et regarder
comment se positionne les villes.


\begin{knitrout}\footnotesize
\definecolor{shadecolor}{rgb}{0.969, 0.969, 0.969}\color{fgcolor}\begin{kframe}
\begin{alltt}
\hlstd{> }\hlstd{med} \hlkwb{<-} \hlstd{med[med}\hlopt{$}\hlstd{spécialité}\hlopt{==}\hlstr{"ophtalmo"}\hlstd{,]}
\hlstd{> }\hlkwd{plot}\hlstd{(}\hlnum{0}\hlstd{,}\hlnum{0}\hlstd{,}\hlkwc{xlim}\hlstd{=}\hlkwd{range}\hlstd{(med}\hlopt{$}\hlstd{Délai),}\hlkwc{ylim}\hlstd{=}\hlkwd{range}\hlstd{(med}\hlopt{$}\hlstd{Tarif),}
\hlstd{+ }     \hlkwc{type}\hlstd{=}\hlstr{"n"}\hlstd{,}\hlkwc{xlab}\hlstd{=}\hlstr{"Délai (jours)"}\hlstd{,} \hlkwc{ylab}\hlstd{=}\hlstr{"Tarif (euros)"}\hlstd{,}
\hlstd{+ }     \hlkwc{main}\hlstd{=}\hlstr{"Ophtalmo"}
\hlstd{+ }     \hlstd{)}
\hlstd{> }\hlkwd{abline}\hlstd{(}\hlkwc{v}\hlstd{=}\hlkwd{mean}\hlstd{(med}\hlopt{$}\hlstd{Délai));} \hlkwd{abline}\hlstd{(}\hlkwc{h}\hlstd{=}\hlkwd{mean}\hlstd{(med}\hlopt{$}\hlstd{Tarif))}
\hlstd{> }
\hlstd{> }\hlkwd{text}\hlstd{(med}\hlopt{$}\hlstd{Délai,med}\hlopt{$}\hlstd{Tarif,med}\hlopt{$}\hlstd{Ville,}\hlkwc{cex}\hlstd{=}\hlnum{0.6}\hlstd{)}
\end{alltt}
\end{kframe}
\end{knitrout}
